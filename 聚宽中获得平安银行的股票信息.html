import seaborn as sns
import matplotlib.pyplot as plt
sns.set()
%matplotlib inline
#plt.style.use('ggplot')

# 获取平安银行近几个工作日的开盘价、最高价、最低价、收盘价。
df = get_price('000001.XSHE',start_date='2020-01-01', end_date='2021-01-01', frequency='daily', fields=['open','high','low','close'])

fg,ax = plt.subplots(2,2,figsize=(16,16))
#sns.kdeplot(df['open'],ax=ax)#曲綫圖



ax[0,0].hist(df['open'],bins=12) # 直方图
ax[0,0].set_title("df[open] hist plot",size=20)

sns.kdeplot(df['open'],ax=ax[0,1])#曲綫圖
ax[0,1].set_title("df[open] kde plot",size=20)

sns.boxplot(data=df,ax=ax[1,0])#箱型圖
ax[1,0].set_title("df box plot",size=20)

sns.violinplot(data=df,ax=ax[1,1])
ax[1,1].set_title("df violin plot",size=20)

plt.tight_layout()
plt.show()
